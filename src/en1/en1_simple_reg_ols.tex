\section{Introduction}

Consider the standard model:
\begin{equation}
Y_t = \alpha x_t + \beta +\epsilon_t
\end{equation}
where available sample is indexed by $t \in {1, 2, \dots, T}$. It is assumed values of independent variable $x$ are assumed to be \textbf{deterministic} i.e. they are imposed by the designer of the experiment.

The assumptions about the error term are:
\begin{equation}
\epsilon_t \sim i.i.d. N(0, \sigma^2)
\end{equation}
Let $\{(x_1, y_1), (x_2, y_2), \dots, (x_T, y_T)\}$ be that sample/dataset s.t.:
\begin{equation}
y_t = \alpha x_t + \beta + e_t
\end{equation}

Minimization of the residual sum of squares:
\begin{equation}
RSS = \sum_{t=1}^{T} e_t^2 = \sum_{t=1}^{T}(y_t - \alpha x_t - \beta)^2
\end{equation}

\section{Derivation of OLS Estimators}

\subsection{Estimators $\hat{\alpha}$ and $\hat{\beta}$}
Partial derivatives of $RSS$:
\begin{eqnarray}
\frac{\partial RSS}{\partial \alpha} &=& -2 \sum_{t=1}^{T}x_t (y_t - \alpha x_t - \beta) \\
\frac{\partial RSS}{\partial \beta} &=& -2 \sum_{t=1}^{T}x_t (y_t - \alpha x_t - \beta)
\end{eqnarray}

Imposing first-order conditions yields and switching to hats (OLS estimators satisfy the FOCs):

\begin{eqnarray}
\sum_{t=1}^{T}x_t (y_t - \hat{\alpha} x_t - \hat{\beta}) &=&  0 \\
\sum_{t=1}^{T} (y_t - \hat{\alpha} x_t - \hat{\beta}) &=& 0
\end{eqnarray}

Note that the second of these conditions can be written briefly as:
\begin{equation}
\overline{y} = \hat{\alpha} \overline{x} + \hat{\beta}
\end{equation}

A little bit of algebra yields:
\begin{equation}
\hat{\alpha} = \frac{\frac{1}{n}\sum\limits_{t=1}^{T}x_t y_t - \overline{x} \overline{y}}
{\frac{1}{n}\sum\limits_{t=1}^{T}x_t^2 - \overline{x}^2}
\end{equation}
